import React from "react";
//Create heading filie
function Heading() {
  let currentHour = new Date().getHours();
  let now = new Date();
  let greeting = "";
  let objectStyle = { color: "" };
  //display greeting based on getHours
  if (currentHour < 12) {
    greeting = "Goodmorning!";
    objectStyle.color = "Red";
  } else if (currentHour < 18) {
    greeting = "Good Afternoon!";
    objectStyle.color = "Green";
  } else {
    greeting = "Good night!";
    objectStyle.color = "blue";
  }
  return (
    <div>
      <h1>It is {now.toLocaleTimeString()}</h1>
      <h1 style={objectStyle}>{greeting}</h1>
    </div>
  );
}
export default Heading;
